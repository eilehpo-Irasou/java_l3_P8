001/*
002Classe pour les items du jeu.
003On peut décrire l'objet
004connaitre son poid, et son prix.*/
005
006public class Items
007{
008        private String Description_objet;
009        private double son_prix;
010        private double son_poids;
011
012        // contructeur de la classe Items
013        /* on utilise le mot clé "final"
014        pour dire qu'un objet ne changera pas
015        de caractéristiques pendant tout 
016        le long du jeu.*/
017        public Items(final String DSO, final double SPX, final double SPS)
018        {
019                this.Description_objet = DSO;
020                this.son_prix = SPX;
021                this.son_poids = SPS;
022        }
023
024        // les méthodes get
025
026        // faire la description d'un objet.
027
028        public String Descripion_item()
029        {
030                return this.Description_objet;
031        }
032
033        // connaitre poids d'un objet
034
035        public double Descripion_poids()
036        {
037                return this.son_poids;
038        }
039
040        // connaitre prix d'un objet
041        public double Descripion_prix()
042        {
043                return this.son_prix;
044        }
045
046// les methodes set
047        public void _DSO(final String DSO)
048        {
049                this.Description_objet=DSO;
050        }
051
052        public void _SPX(final double SPX)
053        {
054                this.son_prix=SPX;
055        }
056
057        public void _SPS(final double SPS)
058        {
059                this.son_poids=SPS;
060        }
061
062
063        public String toString()
064        {
065        return this.Description_objet + "L'item  pèse " 
066        +this.son_prix+"argent et son poids"+this.son_poids+"\n";       
067        }
068
069
070
071}
